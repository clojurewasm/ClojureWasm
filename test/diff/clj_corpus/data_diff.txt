(clojure.data/diff {:a 1 :b 2} {:a 1 :c 3})
;;=> ({:b 2} {:c 3} {:a 1})
(clojure.data/diff [1 2 3] [1 2 4])
;;=> [[nil nil 3] [nil nil 4] [1 2]]
(clojure.data/diff 1 1)
;;=> [nil nil 1]
(clojure.data/diff 1 2)
;;=> [1 2 nil]
(clojure.data/diff {:a 1} {:a 1})
;;=> [nil nil {:a 1}]
(clojure.data/diff {:a {:x 1}} {:a {:x 1 :y 2}})
;;=> (nil {:a {:y 2}} {:a {:x 1}})
(clojure.data/diff [1 2 3] [1 2 3 4])
;;=> [nil [nil nil nil 4] [1 2 3]]
(clojure.data/diff nil nil)
;;=> [nil nil nil]
(clojure.data/diff nil {:a 1})
;;=> [nil {:a 1} nil]
(clojure.data/diff "a" "a")
;;=> [nil nil "a"]
(clojure.data/diff [1 2] [1 2])
;;=> [nil nil [1 2]]
(nth (clojure.data/diff {:a 1 :b 2} {:a 1 :c 3}) 2)
;;=> {:a 1}
(first (clojure.data/diff {:a 1 :b 2} {:b 2}))
;;=> {:a 1}
