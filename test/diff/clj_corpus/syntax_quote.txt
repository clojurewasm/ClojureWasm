`(1 2 ~(+ 1 2))
;;=> (1 2 3)
`[1 ~(* 2 2) 3]
;;=> [1 4 3]
`(~@(list 1 2 3))
;;=> (1 2 3)
`(0 ~@(list 1 2 3) 4)
;;=> (0 1 2 3 4)
(let [x 5] `(~x ~(* x 2)))
;;=> (5 10)
(let [xs [1 2 3]] `[~@xs ~@xs])
;;=> [1 2 3 1 2 3]
`(:k1 :k2 ~(keyword "k3"))
;;=> (:k1 :k2 :k3)
(let [m {:a 1}] `[~@(keys m) ~@(vals m)])
;;=> [:a 1]
`(1 ~@[] 2)
;;=> (1 2)
`(~(inc 0) ~(inc 1) ~(inc 2))
;;=> (1 2 3)
