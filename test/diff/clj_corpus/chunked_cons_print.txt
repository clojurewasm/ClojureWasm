(str (seq (range 3)))
;;=> "(0 1 2)"
(str (rest (range 5)))
;;=> "(1 2 3 4)"
(pr-str [(seq (range 3))])
;;=> "[(0 1 2)]"
(pr-str {:a (seq (range 3))})
;;=> "{:a (0 1 2)}"
(print-str (seq (range 4)))
;;=> "(0 1 2 3)"
(str (seq (range 1000000000 1000000003)))
;;=> "(1000000000 1000000001 1000000002)"
