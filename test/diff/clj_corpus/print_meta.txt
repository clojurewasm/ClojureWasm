(binding [*print-meta* true] (pr-str (with-meta [1 2] {:a 1})))
;;=> "^{:a 1} [1 2]"
(binding [*print-meta* true] (pr-str (with-meta {:x 1} {:m 9})))
;;=> "^{:m 9} {:x 1}"
(pr-str (with-meta [1 2] {:a 1}))
;;=> "[1 2]"
(binding [*print-meta* true] (pr-str [1 2 3]))
;;=> "[1 2 3]"
(binding [*print-meta* true] (pr-str (with-meta (list 1 2) {:k :v})))
;;=> "^{:k :v} (1 2)"
(binding [*print-meta* true] (pr-str [(with-meta [1] {:a 1}) 2]))
;;=> "[^{:a 1} [1] 2]"
(binding [*print-meta* false] (pr-str (with-meta [1] {:a 1})))
;;=> "[1]"
