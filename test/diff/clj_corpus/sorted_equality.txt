(= (sorted-set 1 2 3) #{3 2 1})
;;=> true
(= #{1 2 3} (sorted-set 3 2 1))
;;=> true
(= (sorted-set 1 2) (sorted-set 2 1))
;;=> true
(= (sorted-map :a 1 :b 2) {:b 2 :a 1})
;;=> true
(= {:a 1 :b 2} (sorted-map :b 2 :a 1))
;;=> true
(= (sorted-map :a 1) (sorted-map :a 1))
;;=> true
(= (sorted-set 1 2) #{1 3})
;;=> false
(= (sorted-map :a 1) {:a 2})
;;=> false
(= (sorted-map :a {:x 1}) {:a {:x 1}})
;;=> true
(= (set (sorted-set 1 2 3)) (sorted-set 3 2 1))
;;=> true
