(clojure.string/upper-case "hi")
;;=> "HI"
(clojure.string/lower-case "HI")
;;=> "hi"
(clojure.string/join "," [1 2 3])
;;=> "1,2,3"
(clojure.string/reverse "abc")
;;=> "cba"
(clojure.core/map inc [1 2])
;;=> (2 3)
(clojure.core/filter odd? [1 2 3])
;;=> (1 3)
(clojure.set/union #{1} #{2})
;;=> #{1 2}
(clojure.set/intersection #{1 2} #{2 3})
;;=> #{2}
