(str 100N)
;;=> "100"
(str -7N)
;;=> "-7"
(str 0N)
;;=> "0"
(str 12345678901234567890N)
;;=> "12345678901234567890"
(str 1.5M)
;;=> "1.5"
(str 1.50M)
;;=> "1.50"
(str -3.14M)
;;=> "-3.14"
(str 100M)
;;=> "100"
(str 0.001M)
;;=> "0.001"
(str 0.0000001M)
;;=> "1E-7"
(str 123456789.987654321M)
;;=> "123456789.987654321"
(pr-str 100N)
;;=> "100N"
(pr-str 1.5M)
;;=> "1.5M"
(pr-str 0.001M)
;;=> "0.001M"
(.toString 100N)
;;=> "100"
(.toString 1.5M)
;;=> "1.5"
(str (/ 1 2))
;;=> "1/2"
(str 999999999999999)
;;=> "999999999999999"
(str [100N 1.5M])
;;=> "[100N 1.5M]"
(str {:a 100N})
;;=> "{:a 100N}"
(apply str [1N 2N 3N])
;;=> "123"
(clojure.string/join "," [1N 2M 3N])
;;=> "1,2,3"
