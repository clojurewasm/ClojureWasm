(boolean (re-find #"a(?s:.)b" "a\nb"))
;;=> true
(boolean (re-find #"a(?s:.)b" "a\rb"))
;;=> true
(boolean (re-matches #"(?s).*" "x\ny\nz"))
;;=> true
