(concat [1 2] [3 4])
;;=> (1 2 3 4)
(concat)
;;=> ()
(concat [5])
;;=> (5)
(concat [1] [2] [3])
;;=> (1 2 3)
(concat nil [1] nil [2])
;;=> (1 2)
(concat [] [] [])
;;=> ()
(take 5 (concat [1 2] (range)))
;;=> (1 2 0 1 2)
(apply concat [[1 2] [3] [4 5 6]])
;;=> (1 2 3 4 5 6)
(concat "ab" "cd")
;;=> (\a \b \c \d)
(concat [1 2] '(3 4) (range 5 7))
;;=> (1 2 3 4 5 6)
(concat #{1} [2])
;;=> (1 2)
(into [] (concat (range 3) (range 3)))
;;=> [0 1 2 0 1 2]
(concat {:a 1} {:b 2})
;;=> ([:a 1] [:b 2])
(count (concat (range 100) (range 100) (range 100)))
;;=> 300
(concat nil nil)
;;=> ()
(concat (list 1) (list 2) (list 3) (list 4))
;;=> (1 2 3 4)
