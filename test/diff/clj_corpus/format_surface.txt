(format "%#o" 64)
;;=> "0100"
(format "%#X" 255)
;;=> "0XFF"
(format "% .2f" 3.14)
;;=> " 3.14"
(format "%(.2f" -3.14)
;;=> "(3.14)"
(format "%,.2f" 1234.5)
;;=> "1,234.50"
(format "%+e" 1234.5)
;;=> "+1.234500e+03"
(format "%S" "hi")
;;=> "HI"
(format "%B" true)
;;=> "TRUE"
(format "%-8.3s|" "hello")
;;=> "hel     |"
(format "%8.3s|" "hello")
;;=> "     hel|"
(format "%s" :kw)
;;=> ":kw"
(format "%s" [1 2])
;;=> "[1 2]"
(format "%d" 1000000000000000)
;;=> "1000000000000000"
(format "%,d" 1000000000000000)
;;=> "1,000,000,000,000,000"
(format "%x" -1)
;;=> "ffffffffffffffff"
(format "%10.2e" 1234.5)
;;=> "  1.23e+03"
(format "%d" 1000000000000000)
;;=> "1000000000000000"
(format "%,d" 1000000000000000)
;;=> "1,000,000,000,000,000"
(format "%x" -1)
;;=> "ffffffffffffffff"
(format "%X" 255)
;;=> "FF"
(format "%#x" 255)
;;=> "0xff"
(format "%#X" 255)
;;=> "0XFF"
(format "%#o" 64)
;;=> "0100"
(format "%o" 64)
;;=> "100"
(format "%+.2f" 3.14)
;;=> "+3.14"
(format "% .2f" 3.14)
;;=> " 3.14"
(format "%(.2f" -3.14)
;;=> "(3.14)"
(format "%,.2f" 1234.5)
;;=> "1,234.50"
(format "%+e" 1234.5)
;;=> "+1.234500e+03"
(format "%,.3f" 1234567.891)
;;=> "1,234,567.891"
(format "%S" "hi")
;;=> "HI"
(format "%.3s" "hello")
;;=> "hel"
(format "%-8.3s|" "hello")
;;=> "hel     |"
(format "%8.3s|" "hello")
;;=> "     hel|"
(format "%s" nil)
;;=> "null"
(format "%s" 42)
;;=> "42"
(format "%-10s|" "hi")
;;=> "hi        |"
(format "%05.2f" 3.14159)
;;=> "03.14"
(format "%d" 42)
;;=> "42"
(format "%,d" 1234567)
;;=> "1,234,567"
