(let [m (re-matcher #"\d+" "a12b34")] [(.find m) (.group m)])
;;=> [true "12"]
(let [m (re-matcher #"\d+" "a12b34")] (.find m) [(.start m) (.end m)])
;;=> [1 3]
(let [m (re-matcher #"(\d)(\w)" "1a")] (.find m) [(.group m 0) (.group m 1) (.group m 2) (.groupCount m)])
;;=> ["1a" "1" "a" 2]
(.matches (re-matcher #"\d+" "123"))
;;=> true
(.matches (re-matcher #"\d+" "12a"))
;;=> false
(.lookingAt (re-matcher #"\d+" "123abc"))
;;=> true
(.lookingAt (re-matcher #"\d+" "abc123"))
;;=> false
(let [m (re-matcher #"\d" "1 2 3")] [(.find m) (.find m) (.find m) (.find m)])
;;=> [true true true false]
(let [m (re-matcher #"\d+" "x99")] (.find m) (.reset m) (.find m))
;;=> true
(re-find (re-matcher #"\d+" "abc123"))
;;=> "123"
(re-groups (doto (re-matcher #"(\d)(\d)" "12") .find))
;;=> ["12" "1" "2"]
