(Boolean/parseBoolean "true")
;;=> true
(Boolean/parseBoolean "false")
;;=> false
(Boolean/parseBoolean "TRUE")
;;=> true
(Boolean/valueOf "true")
;;=> true
(Boolean/valueOf true)
;;=> true
(Boolean/toString true)
;;=> "true"

;;=> false
(Boolean/logicalOr true false)
;;=> true
(Boolean/logicalXor true true)
;;=> false
Boolean/TRUE
;;=> true
